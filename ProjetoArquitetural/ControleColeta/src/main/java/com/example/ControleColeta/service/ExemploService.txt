package corp.ambiental.middleware.services;

import corp.ambiental.middleware.model.dto.CidadeDTO;
import corp.ambiental.middleware.repository.CidadeRepository;
import corp.ambiental.utils.Messages;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.core.convert.ConversionService;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.stream.Collectors;

import static corp.ambiental.utils.MessagesConstantes.PADRAO_BUSCANDO_TODAS_DEBUG;

@Service
public class CidadeService {

    public static final Logger log = LoggerFactory.getLogger(CidadeService.class);

    @Autowired
    private CidadeRepository cidadeRepository;
    @Autowired
    private ConversionService conversionService;
    @Autowired
    private Messages messages;

    public List<CidadeDTO> recuperarCidades() {
        log.info(messages.get(PADRAO_BUSCANDO_TODAS_DEBUG), CidadeDTO.class);
        return cidadeRepository.findAll()
                .stream()
                .map(cidade -> conversionService.convert(cidade, CidadeDTO.class))
                .collect(Collectors.toList());
    }
}
