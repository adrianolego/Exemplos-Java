package corp.ambiental.middleware.repository;

import corp.ambiental.middleware.model.entity.AtributoValor;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;
import org.springframework.stereotype.Repository;

import java.util.List;
import java.util.Map;

@Repository
public interface AtributoValorRepository extends JpaRepository<AtributoValor, Long> {

    @Query(value = " SELECT AV.idValor, AV.descricao FROM AtributoValor AV WHERE AV.atributo.id = :idAtributo ")
    List<Map<String, Object>> obterAtributoValor(@Param("idAtributo") Long idAtributo);
}
